SELECT MEAL_NAME, COUNT(*) 
FROM ORDERS
GROUP BY MEAL_NAME;

SELECT 'TOTAL' as MEAL_NAME, SUM(PRICE) as TOTAL_PRICE
FROM MEAL 
WHERE PRICE > 15
GROUP BY 'TOTAL';

SELECT MEAL_NAME, COUNT(*) 
FROM MEALS
WHERE CALORIES > 500
GROUP BY MEAL_NAME;

SELECT SUM(PRICE) as TOTAL_PRICE
FROM MEALS
WHERE LOWER(NAME) LIKE '%s%';

SELECT MEAL_NAME, COUNT(*)
FROM MEAL_ORDER
WHERE TIME_MEAL_ORDERED < '2023-10-10 19:23:00' 
GROUP BY MEAL_NAME;
